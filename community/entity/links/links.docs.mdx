import { Links } from './links';

This component describes the Link object. It serves as the source of truth for components providing or consuming Links.

### Create a new link object

```ts
import { Links } from '@teambit/community.entity.links';

const myLink = new Links('https://bit.dev', 'Bit');
```

### Consume Links via an API

```ts
import type { LinksFromApi } from '@teambit/community.entity.links';
import { Links } from '@teambit/community.entity.links';
import { fetchLink } from './fetch-link';

async () => {
  const fetchedLink: LinksFromApi = await fetchLink();
  const link = Links.fromApiObject(fetchedLink);
};
```
