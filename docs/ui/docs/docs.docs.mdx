import { RoutingProvider } from '@teambit/base-ui.routing.routing-provider';
import { useLocation, MemoryRouter } from 'react-router-dom';
import { Link } from '@teambit/ui-foundation.ui.react-router.link';
import { NavLink } from '@teambit/ui-foundation.ui.react-router.nav-link';
import { Docs } from './docs';

`Docs` is a component-based documentation section built in React. It is a React component you can use in your website's components.

The 'Docs' component renders the docs page (MDX) with a sidebar to its left.  
This component receives and renders all the Routes and makes them available in ReactRouter.

### Component usage

```js live
() => {
  const routing = { Link, NavLink, useLocation };
  const primaryRoutes = [
    {
      path: 'quick-start',
      icon: 'lightning',
      title: 'Quick Start',
      description: 'Quick Start',
      component: <div>My quick start content...</div>,
    },
    {
      path: 'thinking-in-components',
      icon: 'Lightbulb-thinking',
      title: 'Thinking in components',
      component: <div>Thinking in components contnet</div>,
    },
  ];
  const routes = [
    {
      path: 'installation',
      title: 'Installation',
      component: <div />,
      children: [
        {
          path: 'start-bit-project',
          title: 'Start a new Bit Workspace',
          component: <div>start bit project contnet</div>,
        },
      ],
    },
    {
      path: 'components',
      title: 'Components',
      component: <div />,
      children: [
        {
          path: 'creating-components',
          title: 'Creating components',
          component: <div>creating compoents contnet</div>,
        },
        {
          path: 'component-config',
          title: 'Component Configuration',
          component: <div>component config</div>,
        },
      ],
    },
  ];

  return (
    <RoutingProvider value={routing}>
      <MemoryRouter>
        <Docs baseUrl="/" contents={[{ routes }]} primaryLinks={primaryRoutes} />
      </MemoryRouter>
    </RoutingProvider>
  );
};
```
