### Component usage

```js

```

### Render with theme and MDX providers

```js
import { ThemeCompositions } from '@teambit/documenter.theme.theme-compositions';
import { MDXLayout } from '@teambit/mdx.ui.mdx-layout';
import { ImportComponents } from './index';

<ThemeCompositions>
  <MDXLayout>
    <ImportComponents />
  </MDXLayout>
</ThemeCompositions>
```
