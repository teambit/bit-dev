import { MemoryRouter, UNSAFE_LocationContext } from 'react-router-dom';
import { Link, NavLink } from '@teambit/base-react.navigation.link';
import { RouterProvider, useLocation } from '@teambit/base-react.navigation.router-context';
import { reactRouterAdapter } from './react-routing-provider';

A React Router adapter for our routing system. Any `<Link>` rendered under this will have the React Router implementation!

Simply wrap you app with `reactRouterAdapter`, like so:

```tsx
function AppContext({ children }) {
  return <RouterProvider value={reactRouterAdapter}>{children}</RouterProvider>;
}
```

Try it out!

```tsx live
() => {
  return (
    <UNSAFE_LocationContext.Provider value={null}> {/* <-- used to allow nested router, not needed in your app */}
      <MemoryRouter> {/* <-- required by react-router*/}
        <RouterProvider value={reactRouterAdapter}>
          <CurrentLocation />
          <NavLink href="/some/link" activeStyle={{ fontWeight: 'bold' }}>
            /some/link
          </NavLink>
          <br />
          <NavLink href="/another/link" activeStyle={{ fontWeight: 'bold' }}>
            /another/link
          </NavLink>
        </RouterProvider>
      </MemoryRouter>
    </UNSAFE_LocationContext.Provider>
  );

  function CurrentLocation() {
    const location = useLocation();

    return <div>current location is: {JSON.stringify(location, undefined, 1)}</div>;
  }
};
```
