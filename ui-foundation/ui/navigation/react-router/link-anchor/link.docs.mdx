import { LinkContextProvider } from './link-context';

> ⚠️ Internal! Do not use!

A wrapper for React Router LinkAnchor, that adds the additional capability of a baseUrl.  
We use it to manipulate the links in the DOM, without effecting actual navigation.

```tsx live
<LinkContextProvider baseUrl="https://bit.dev">
  <LinkAnchor navigate={() => {}} href="/some-link">
    /some-link
  </LinkAnchor>
  <div>Check the href written in the DOM</div>
</LinkContextProvider>
```
